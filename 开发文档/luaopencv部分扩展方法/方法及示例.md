local dst = "/sdcard/target.png"
local bExit = false

function onClose()
	bExit = true
end

function showImage(path)
	ui.newLayout("layout1")
	ui.addImageView("layout1","imgId1",path)
	ui.setOnClose("layout1","onClose()")
	ui.show("layout1")
	while bExit == false do
		sleep(100)
	end
end

local screen = cv.snapShot(0,0,0,0) --大图

local image = cv.snapShot(200,100,300,300) --截取一张要查找的小图

local result = cv.Mat.new()

cv.matchTemplate(screen, image,result, cv.TM_CCOEFF_NORMED)

local new_minLoc = cv.newPoint(0,0)
local new_maxLoc = cv.newPoint(0,0)
local new_minValue = cv.newDouble(0.0)
local new_maxValue = cv.newDouble(0.0)

cv.minMaxLoc(result,new_minValue,new_maxValue,new_minLoc,new_maxLoc)

local pos = cv.getPoint(new_maxLoc)

local matchRect = cv.Rect(pos.x,pos.y,image.cols,image.rows)

cv.rectangle(screen, matchRect, cv.Scalar(0, 255, 0), 2);

cv.imwrite(dst,screen)

showImage(dst)

image:release()--释放Mat
screen:release()--释放Mat

if new_minLoc ~= nil then
	cv.deletePtr(new_minLoc)
end
if new_maxLoc ~= nil then
	cv.deletePtr(new_maxLoc)
end
if new_minValue ~= nil then
	cv.deletePtr(new_minValue)
end
if new_maxValue ~= nil then
	cv.deletePtr(new_maxValue)
end

local dst = "/sdcard/target.png"
function onClose()
exitScript()
end

function showImage(path)
	ui.newLayout("layout1")
	ui.addImageView("layout1","imgId1",path)
	ui.setOnClose("layout1","onClose()")
	ui.show("layout1")
	sleep(100000)

--cv.imread(src, cv.IMREAD_COLOR)
local image = cv.snapShot(0,0,500,500)
if image ~= nil then
	local angle = 45.0 -- 旋转角度
	local center = cv.Point2f(image.cols / 2, image.rows / 2) -- 旋转中心

	local rotMatrix = cv.getRotationMatrix2D(center, angle, 1.0) -- 获取旋转矩阵
	local out = cv.Mat.new(image.cols,image.rows)
	cv.warpAffine(image, out,rotMatrix, cv.Size(image.cols, image.rows))
	cv.imwrite(dst, out)
	showImage(dst)
	image:release()--释放Mat
	out:release()--释放Mat
end
collectgarbage("collect")
sleep(100)
